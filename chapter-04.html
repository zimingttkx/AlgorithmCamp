chapter-04 网格图

一、算法思路和模板

1.1 网格图DFS模板（计算连通块大小）

// 网格图DFS模板 - 计算连通块大小
int m = grid.size(), n = grid[0].size();
int area = 0;
function dfs(int x, int y) {
    if (x < 0 || x >= m || y < 0 || y >= n || grid[x][y] != 1) return;
    grid[x][y] = 0;  // 标记已访问
    area++;
    dfs(x + 1, y);
    dfs(x - 1, y);
    dfs(x, y + 1);
    dfs(x, y - 1);
}

// 遍历所有格子
for (int i = 0; i < m; i++) {
    for (int j = 0; j < n; j++) {
        if (grid[i][j] == 1) {
            area = 0;
            dfs(i, j);
            // area 就是连通块的大小
        }
    }
}

1.2 网格图BFS模板（单源最短路）

// 网格图BFS模板 - 单源最短路
vector<vector<int>> dist(m, vector<int>(n, -1));
queue<pair<int, int>> q;
q.emplace(sx, sy);
dist[sx][sy] = 0;

while (!q.empty()) {
    auto [x, y] = q.front();
    q.pop();
    // 如果提前找到目标，可以在这里返回
    // if (x == tx && y == ty) return dist[x][y];
    for (auto& [nx, ny] : neighbors(x, y)) {
        if (nx >= 0 && nx < m && ny >= 0 && ny < n && dist[nx][ny] == -1) {
            dist[nx][ny] = dist[x][y] + 1;
            q.emplace(nx, ny);
        }
    }
}

1.3 二叉树DFS与网格图DFS的区别

区别 二叉树DFS 网格图DFS
方向 左右两个方向 上下左右四个方向
visited处理 通常不需要（无环） 必须标记已访问（可能有环）
边界条件 node == nullptr 越界或不符合条件
